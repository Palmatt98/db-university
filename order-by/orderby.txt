# 1 Contare quanti iscritti ci sono stati ogni anno
SELECT `students`.`enrolment_date`,COUNT( `id`)
FROM `students` 
GROUP BY `students`.`enrolment_date`

# 2 Contare gli insegnanti che hanno l'ufficio nello stesso edificio
SELECT `teachers`.`name`AS `teacher_name` ,`teachers`.`surname` AS `teacher_surname`,`teachers`.`office_address`
FROM `teachers`
ORDER BY `teachers`.`office_address`
